[子集](https://leetcode.cn/problems/subsets/)

## 解法
```js
var subsets = function(nums) {
    function dfs(cur, index) {
        res.push(cur)
        console.log(cur)
        for (let i = index; i < nums.length; i++) {
            dfs([...cur, nums[i]], i+1)
        }
    }

    let res = [];
    dfs([], 0)
    return res
};
```
